# The any() function returns True if any item in an iterable are true, otherwise it returns False.
# If the iterable object is empty, the any() function will return False.
# example of any function
mylist = [False, True, False]
x = any(mylist)
print(x)


inputt =input()
print (any(c.isalnum() for c in inputt))
print (any(c.isalpha() for c in inputt ))
print( any(c.isdigit() for c in inputt))
print( any(c.islower() for c in inputt))
print (any(c.isupper() for c in inputt))
